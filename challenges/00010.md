size_t receive_line(int fd, char *buf, size_t size) {
    char c;
    size_t total = 0;
    size_t read = 0;
    while (1) {
        if (receive(fd, &c, 1, &read) != 0) {
            _terminate(1);
        }
        if (read == 0) {
            return 0;
        }
        if (total >= size) {
            return 0;
        }
        buf[total++] = c;
        if (c == '\n') {
            break;
        }
    }
    return total;
}
void *copy(void *dst, const void *src, size_t size) {

    while (size--)

    return dst;
}
